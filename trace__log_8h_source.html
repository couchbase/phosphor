563  inline ChunkTenant* getChunkTenant() {
564  auto shared_index =
565  platform::getCurrentThreadIDCached() % shared_chunks.size();
566 
567  ChunkTenant& cs = (thread_chunk.sentinel)
568  ? thread_chunk
569  : shared_chunks[shared_index];
570 
571  if (!cs.sentinel->acquire()) {
572  resetChunk(cs);
573  return nullptr;
574  }
575  // State is busy
576  if (!cs.chunk || cs.chunk->isFull()) {
577  replaceChunk(cs);
578  if (!cs.chunk) {
579  return nullptr;
580  }
581  }
582 
583  return &cs;
584  }
